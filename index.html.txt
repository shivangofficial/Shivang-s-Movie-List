<script>
/* ------- LOCAL STORAGE HANDLING ------- */
function loadMovies() {
  return JSON.parse(localStorage.getItem("movies") || "[]");
}
function saveMovies(movies) {
  localStorage.setItem("movies", JSON.stringify(movies));
}

/* ------- INITIAL RENDER ------- */
let movies = loadMovies();
let editIndex = null;

function renderMovies() {
  const grid = document.getElementById("movieGrid");
  grid.innerHTML = "";

  movies.forEach((movie, i) => {
    const card = document.createElement("div");
    card.className = "movie-card";
    card.onclick = () => openEditModal(i);

    card.innerHTML = `
      <img class="movie-poster" src="${movie.poster || ""}">
      <div class="movie-info">
        <div class="movie-title">${movie.title}</div>
        <div class="movie-year">${movie.year}</div>
      </div>
    `;

    grid.appendChild(card);
  });
}
renderMovies();

/* ------- MODAL LOGIC ------- */
function openAddModal() {
  editIndex = null;

  document.getElementById("modalTitle").innerText = "Add Movie";
  document.getElementById("titleInput").value = "";
  document.getElementById("yearInput").value = "";
  document.getElementById("directorInput").value = "";
  document.getElementById("writerInput").value = "";
  document.getElementById("posterInput").value = "";
  document.getElementById("notesInput").value = "";

  document.getElementById("modalOverlay").style.display = "flex";
}

function openEditModal(i) {
  editIndex = i;

  document.getElementById("modalTitle").innerText = "Edit Movie";

  let m = movies[i];
  document.getElementById("titleInput").value = m.title;
  document.getElementById("yearInput").value = m.year;
  document.getElementById("directorInput").value = m.director;
  document.getElementById("writerInput").value = m.writer;
  document.getElementById("posterInput").value = m.poster;
  document.getElementById("notesInput").value = m.notes;

  document.getElementById("modalOverlay").style.display = "flex";
}

function closeModal() {
  document.getElementById("modalOverlay").style.display = "none";
}

function saveMovie() {
  const movie = {
    title: document.getElementById("titleInput").value,
    year: document.getElementById("yearInput").value,
    director: document.getElementById("directorInput").value,
    writer: document.getElementById("writerInput").value,
    poster: document.getElementById("posterInput").value,
    notes: document.getElementById("notesInput").value
  };

  if (editIndex !== null) movies[editIndex] = movie;
  else movies.push(movie);

  saveMovies(movies);
  renderMovies();
  closeModal();
}
